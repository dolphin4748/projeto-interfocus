create sequence clientes_seq;
create sequence dividas_seq;
create sequence contas_seq;

create table clientes (
	id bigint not null default nextval('clientes_seq'),
	nome varchar(100),
	cpf varchar(20),
	email text,
	datanascimento date,
	datacadastro timestamp not null default now(),
	primary key (id)
);

create table dividas (
	id bigint not null default nextval('dividas_seq'),
	valor integer not null,
	descricao text,
	datacadastro timestamp not null default now(),
	primary key (id)
);

create table contas (
	id bigint not null default nextval('contas_seq'),
	clientesid bigint,
	dividasid bigint,
	dataconta timestamp not null default now(),
	primary key (id),
	FOREIGN KEY (clientesid) REFERENCES clientes(id),
	FOREIGN KEY (dividasid) REFERENCES dividas(id)
);
